def normalise(dataframe: pd.DataFrame):
    scaler = MinMaxScaler()
    scaler.fit(dataframe)
    scaled_dataframe = scaler.transform(dataframe)
    return scaler, scaled_dataframe

def extrapolate_context_vector(cj: np.array, ck: np.array, lambda_: float):
    extrapolated_cj = (cj - ck) * lambda_ + cj
    return extrapolated_cj

            drop_cols.append(col)

df = df.drop(['inputs_missing_values'], axis=1)

df = df.replace(-1, np.nan)

print('Total des valeurs manquantes avant interpolation: {}'.format(missing_value_count_before))
for col in df:

missing_value_count_after = df.isna().sum().sum()

# Encoder
encoder_input = Input(shape=(timesteps, n_features))
lstm_0 = LSTM(SEQ_1_NODES, activation='relu', return_sequences=True)(encoder_input)
dropout_0 = Dropout(DROPOUT_RATE)(lstm_0)
lstm_1 = LSTM(LATENT_DIM, activation='relu')(dropout_0)
encoded_output = Dropout(DROPOUT_RATE)(lstm_1)

encoder = Model(encoder_input, encoded_output)
    
# Decoder
decoder_input = Input(shape=(LATENT_DIM))
repeat_layer = RepeatVector(timesteps)(decoder_input)
lstm_2 = LSTM(SEQ_1_NODES, activation='relu', return_sequences=True)(repeat_layer)
dropout_2 = Dropout(DROPOUT_RATE)(lstm_2)
lstm_3 = LSTM(LATENT_DIM, activation='relu', return_sequences=True)(dropout_2)
dropout_3 = Dropout(DROPOUT_RATE)(lstm_3)
decoder_output = TimeDistributed(Dense(n_features))(dropout_3)

decoder = Model(decoder_input, decoder_output)

# Implmentation de l'autoencoder
input_layer = Input(shape=(timesteps, n_features))
encoded = encoder(input_layer)
decoded = decoder(encoded)
autoencoder = Model(input_layer, decoded)
autoencoder.compile(optimizer=Adam(learning_rate=1e-3), loss='mse')
